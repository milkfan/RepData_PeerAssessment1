Imputing missing values

sum(is.na(activity$steps))

## [1] 2304

activity_fill <- data.frame(activity)
for (i in 1:dim(activity)[1]) {
    if (is.na(activity$steps[i])) {
        activity_fill$steps[i] <- interval_steps[as.character(activity$interval[i])]
    }
}
